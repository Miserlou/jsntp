jsntp

jsntp lets you synchronize clocks through a web browser using socket.io; it's Network Time Protocol for the browser in pure Javascript. An example:

The demo requires Javascript
Synchronizing...

It comes with at.js, which lets you schedule tasks to happen at certain times.

Howto

Client

<script src="/socket.io/socket.io.js"></script>
<script src="/ntp/ntp.js"></script>
<script src="/ntp/at.js"></script>
<script>
ntp.sync(function(){
  alert(
    'Your clock differs from the server\'s'
  + ' by '+(ntp.offset/1000)+' seconds.'
  );
  at.at(function(){
    alert('Happy New Year!');
  },ntp.date(1325376000000));
});
</script>

Server

ntp=require('ntp');
var app = require('http').createServer(handler) 
app.listen(1337);
ntp.listen(app);

function handler (request,response) {
  ntp.static(request,response)
}
