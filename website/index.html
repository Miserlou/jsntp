jsntp

jsntp lets you synchronize clocks through a web browser using socket.io; it's Network Time Protocol for the browser in pure Javascript.

Demo

Some sort of demo. Maybe just a clock.

Howto

Client

<script src="/socket.io/socket.io.js"></script>
<script src="/ntp/ntp.js"></script>
<script src="/ntp/at.js"></script>
<script>
ntp.sync(function(){
  alert(
    'Your clock differs from the server\'s'
  + ' by '+(ntp.offset/1000)
  );
  at.at(function(){
    alert('Happy New Year!');
  },ntp.date(1325376000000));
});
</script>

Server

ntp=require('ntp');

function handler (request,response) {
  ntp.static(request,response)

  /*
     Your code goes below
     In this case, we're just loading the demo.
  */
  require('fs').readFile('demo.html',
    function(error, content) {
      if (error) {
        response.writeHead(500);
        response.end();
      } else {
        response.writeHead(200, {
          'Content-Type': 'text/html'
        });
        response.end(content, 'utf-8');
      }
    });
  /*
     End of your code
  */
}
